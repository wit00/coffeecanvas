The triangle creates a triangle using 3 points. The triangle is also one of my graphical primitives created using the standard functions.

 
class Triangle extends Shape
  constructor:(args = {})-> 
    super(args.x,args.y,args.rotationVelocity,args.velocity,args.otherVelocity,
    args.alpha,args.fill,args.stroke,args.lineCap,args.lineWidth,args.lineJoin,args.miterLimit)
    @x1 = args.x1 or 10
    @y1 = args.y1 or 10
    @x2 = args.x2 or 0
    @y2 = args.y2 or 20
    @x3 = args.x3 or 20
    @y3 = args.y3 or 20
    @x = @x1
    @y = @y1
  draw: (context) =>
    context.fillStyle = @fill
    context.strokeStyle = @stroke
    context.beginPath()
    context.moveTo(0,0)
    context.lineTo(@x2-@x1,@y2-@y1)
    context.lineTo(@x3-@x1,@y3-@y1)
    context.lineTo(0,0)
    context.stroke()
    context.fill()
    context.closePath()
