curve.coffee

¶

The curve class draws a quadratic or cubic bezier curve depending on the arguments supplied. The default is a quadratic curve, but if two control points with both x and y coordinates are given, then a cubic curve is drawn. Curve also has an endpoint that is default at (100,100)

 
class Curve extends Shape
   constructor:(args = {})-> 
    super(args.x,args.y,args.rotationVelocity,args.velocity,args.otherVelocity,
    args.alpha,args.fill,args.stroke,args.lineCap,args.lineWidth,args.lineJoin,args.miterLimit)
    @cp1x = args.controlX1 or 0
    @cp1y = args.controlY1 or 0
    @cp2x = args.controlX2 or "none"
    @cp2y = args.controlY2 or "none"
    @endX = args.endX or 100
    @endY = args.endY or 100 
    
   draw: (context) =>
    context.strokeStyle = @stroke
    context.beginPath()
    context.moveTo(@x,@y)
    if @cp2x and @cp2y isnt "none"
      context.bezierCurveTo(@cp1x,@cp1y,@cp2x,@cp2y,@endX,@endY) 
    else
      context.quadraticCurveTo(@cp1x,@cp1y,@endX,@endY)
    context.stroke()
    context.closePath()
