class AnimationManager
  constructor: ->
    @sprites = []
    @lastTime = new Date()
    @thisTime = new Date()
  onFrameEvent: () -> 
    @thisTime = new Date()
    animationEvent = {type:"animation",keyCode:0}
    sprite.receiveEvent(animationEvent,true,@thisTime.getTime()-@lastTime.getTime()) for sprite in @sprites
    tweenEvent = {type:"tween",keyCode:0}
    sprite.receiveEvent(tweenEvent,true,@thisTime.getTime()-@lastTime.getTime())  for sprite in @sprites
    @lastTime = @thisTime
  observe: (sprite) ->
    @sprites.push(sprite)
