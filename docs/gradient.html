class Gradient
  constructor:(args={})-> 
    @x1 = args.x1 or 0
    @y1 = args.y1 or 0
    @x2 = args.x2 or 60
    @y2 = args.y2 or 0
    @r1 = args.r1 or 5
    @r2 = args.r2 or 60
    @colorStops = args.colorStops or [["red",1],["blue",0],["white",0.5]]

  makeOptimalHorizontal:(sprite)->
    @y1 = 0
    if sprite.radius
      @x1 = 0  - sprite.radius
      @x2 = sprite.radius
    if sprite.width
      @x1 = 0
      @x2 = sprite.width
    @y2 = 0

  makeOptimalVertical:(sprite)->
    @x1 = 0
    @y1 = 0
    @x2 = 0
    if sprite.radius
      @y2 = sprite.radius/2
    if sprite.width
      @y2 = sprite.width
    
  makeOptimalRadial:(sprite)->
    @x1 = 0
    @y1 = 0
    @x2 = 0
    @y2 = 0
    if sprite.radius
      @r1 = sprite.radius/10
      @r2 = sprite.radius
    if sprite.width
      @r1 = sprite.width/10
      @r2 = sprite.width
