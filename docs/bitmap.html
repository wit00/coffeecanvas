Bitmap takes draws a bitmap, png, jpg, or svg file that is loaded into the html. Bitmap requires the image id from the html.

 
class Bitmap extends Shape
  constructor:(args = {})-> 
    super(args.x,args.y,args.rotationVelocity,args.velocity,args.otherVelocity,
    args.alpha,args.fill,args.stroke,args.lineCap,args.lineWidth,args.lineJoin,args.miterLimit)
    @image = document.getElementById(args.imgId);
  draw: (context) =>
    context.drawImage(@image,@x,@y)
